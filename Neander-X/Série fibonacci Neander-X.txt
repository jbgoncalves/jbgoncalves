;Faça um programa que, dado um número N, diga qual valor da sequência de Fibonacci aquele número é.

LDI 1     ;Carrego o valor 1 no ACC
STA 170   ;Armazeno o valor do ACC na memória. POS 170, ATUAL
STA 180   ;Armazeno o valor do ACC na memória. POS 180, SUBDOCONT
IN 0      ;Armazeno valor do buffer do teclado no ACC
SUB 180   ;Subtraio o indice do USER, já que ele vai parar em 0
JZ 38     ;Usuario digita 1, fazendo com q nao saia do laço no JNZ, pois 0-1 é -1
STA 150   ;Armazeno valor do ACC na memória. POS 150, USER

LDI 0     ;Armazeno o valor 0 no ACC
STA 160   ;Armazeno o valor do ACC na memória. POS 160, ANT. 1 + 0 primeira passada


LDA 170   ;Carrego valor do ATUAL no ACC
ADD 160   ;Somo valor do ACC(ATUAL) + ANT
JN 43     ;Se o resultado da soma do ATUAL + ANT deu negativo, ocorreu overflow, retorno FF(255) para o usuario perceber isso
STA 170   ;Armazeno o valor do ACC em ATUAL
SUB 160   ;Subtraio o valor do ACC(ATUAL) - ANT(novo ant)
STA 160   ;Armazeno novo ANT

LDA 150   ;Carrego o valor do USER no ACC
SUB 180   ;-- do índice do USER
STA 150   ;Armazeno o valor do USER na memória
         
JNZ 18    ;Se o resultado da SUBTRAÇÃO deu != 0, seguir fazendo as operações novamente 

LDA 170   ;Carrego o valor do ATUAL no ACC para mostrá-lo
OUT 0     ;Mostro no visor o resultado de ATUAL
HLT       ;Encerro o programa

LDI 255  ;Tratamento para overflow
OUT 0     ;Mostro no visor o overflow FF(255)
HLT       ;Encerro o programa
